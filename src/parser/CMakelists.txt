
cmake_minimum_required(VERSION 3.16)

# 1. CREATE the target first
add_library(parser
    AnonLabels.cpp
    AnonLabels.h
    ANSI_esc.h
    ASTNode.cpp
    ASTNode.h
    ast_source_extractor.h
    common_types.h
    expressionparser.cpp
    expressionparser.h
    expr_rules.cpp
    expr_rules.h
    expr_tokens.cpp
    expr_tokens.h
    grammar_rule.cpp
    grammar_rule.h
    handle_binary_op.h
    opcodedict.cpp
    opcodedict.h
    parser.cpp
    parser.h
    sym.h
    symboltable.cpp
    symboltable.h
    token.h
    tokenizer.cpp
    tokenizer.h
    utils.cpp
    utils.h
)

# 2. NOW you can apply properties to it

target_compile_options(parser PRIVATE 
    $<$<OR:$<CXX_COMPILER_ID:GNU>,$<CXX_COMPILER_ID:Clang>>:-O3 -march=native>
    $<$<CXX_COMPILER_ID:MSVC>:/O2> # Optional: Ensures Windows Release is also fast
)
target_compile_definitions(parser PUBLIC USE_BOOST_REGEX)
target_include_directories(parser PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(parser PRIVATE Boost::regex)
