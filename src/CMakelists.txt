cmake_minimum_required(VERSION 3.16)

add_executable(pasm main.cpp)
target_compile_options(pasm PRIVATE 
    $<$<OR:$<CXX_COMPILER_ID:GNU>,$<CXX_COMPILER_ID:Clang>>:-O3 -march=native>
    $<$<CXX_COMPILER_ID:MSVC>:/O2> # Optional: Ensures Windows Release is also fast
)

# This adds the 'parser' subdirectory
add_subdirectory(parser)

# This links pasm to parser AND automatically inherits 
# the include directories and definitions from 'parser'
target_link_libraries(pasm PUBLIC parser)

# No need for manual target_include_directories(pasm...) here 
# if 'parser' uses 'target_include_directories(... PUBLIC ...)'
